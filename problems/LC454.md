# 454. 4Sum II

## Solution: hashmap + loop

TC: O(n^2)

```c++
int fourSumCount(vector<int>& nums1, vector<int>& nums2, vector<int>& nums3, vector<int>& nums4) {
    unordered_map<int, int> freq;
    int n = nums1.size(), cnt = 0;
    for (int i = 0;i<n;i++) {
        for (int j = 0;j<n;j++) {
            freq[nums1[i]+nums2[j]]++;
        }
    }
    for (int i = 0;i<n;i++) {
        for (int j = 0;j<n;j++) {
            int sum = nums3[i]+nums4[j];
            cnt += freq[-sum];
        }
    }
    return cnt;
}
```
