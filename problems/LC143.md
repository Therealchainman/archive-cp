Solution: Stack

void reorderList(ListNode* head) {
    stack<ListNode*> stk;
    ListNode *cur = head;
    while (cur) {
        stk.push(cur);
        cur = cur->next;
    }
    for (int i = stk.size()/2;i>0;i--) {
        ListNode *fromStack = stk.top();
        stk.pop();
        ListNode *tmp = head->next;
        head->next = fromStack;
        head = head->next;
        head->next = tmp;
        head=head->next;
    }
    head->next = nullptr;
}
