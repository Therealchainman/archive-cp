## B. Supply Chain

### Solution 1: max element in array, sum

Observe that the difference between all the snowballs creation time is fixed between each one and that it is going to be equal to the slowest processing time. 

The slowest processing time means that is the bottleneck and everything will be now x time split. 

Given it is fixed you just need to find the first time at which a snowball is created, and then go from there. 

```cpp
int N, T;
vector<int> A;

void solve() {
    cin >> N >> T;
    A.resize(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    int delta = *max_element(A.begin(), A.end());
    int64 start = accumulate(A.begin(), A.end(), 0LL);
    if (start > T) {
        cout << 0 << endl;
        return;
    }
    int ans = (T - start) / delta + 1;
    cout << ans << endl;
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    solve();
    return 0;
}
```

## C. Frosted Highway

### Solution 1: prefix sum, gcd

Need a prefix sum for fast range queries to determine the number of elements with gcd = 1.

```cpp
int N, Q;
vector<int> psum;

void solve() {
    cin >> N >> Q;
    psum.assign(N + 1, 0);
    for (int i = 0; i < N; ++i) {
        int x;
        cin >> x;
        psum[i + 1] += psum[i];
        if (gcd(i + 1, x) == 1) psum[i + 1]++;
    }
    while (Q--) {
        int l, r;
        cin >> l >> r;
        int ans = psum[r] - psum[l - 1];
        cout << ans << endl;
    }
}
```

## D. Snowball

### Solution 1: prefix sum, sorting, suffix sum

Key realization is that the best answer is going to have to be at one of the times when skeletons spawn.

And you can loop through those spawn events in order, increasing, and along the way keep track of the prefix sum and suffix sum.  

prefix sum tracks the amount of damage all the skeletons spawned prior do, and suffix sum the damage all skeletons afterwards will do

```cpp
const int64 INF = numeric_limits<int64>::max();
int N, K;
vector<int> A, coords;
map<int, int> freq;

void solve() {
    cin >> N >> K;
    A.resize(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
        coords.emplace_back(A[i]);
        freq[A[i]]++;
    }
    sort(coords.begin(), coords.end());
    coords.erase(unique(coords.begin(), coords.end()), coords.end());
    int M = coords.size();
    vector<int> F;
    for (const auto &[k, v]: freq) {
        F.emplace_back(v);
    }
    vector<int64> ssum(M + 1, 0);
    for (int i = M - 1; i >= 0; --i) {
        ssum[i] += ssum[i + 1] + 1LL * (K - coords[i]) * F[i];
    }
    int leftCnt = 0, prv = 0;
    int64 ans = INF, psum = 0;
    for (int i = 0; i < M; ++i) {
        int delta = coords[i] - prv;
        psum += 1LL * leftCnt * delta;
        int64 cand = psum + ssum[i + 1];
        ans = min(ans, cand);
        leftCnt += F[i];
        prv = coords[i];
    }
    cout << ans << endl;
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    solve();
    return 0;
}
```

## F. Frosted Highway Hard

### Solution 1: harmonic loop, binary search

The key is you can use a harmonic loop which has about O(NlogN) time complexity for figuring out for each k what indexes match the gcd. 

By storing the indexes in a lookup table, then can query the lower and upper bound to find the number of index where that gcd was divisible by k.

```cpp
const int MAXK = 1e3;
int N, Q;
vector<int> A;
vector<vector<int>> lookup;

void solve() {
    cin >> N >> Q;
    A.resize(N + 1);
    for (int i = 1; i <= N; ++i) {
        cin >> A[i];
    }
    lookup.resize(MAXK + 1);
    for (int k = 1; k <= MAXK; ++k) {
        for (int i = k; i <= N; i += k) {
            if (gcd(i, A[i]) % k == 0) lookup[k].emplace_back(i);
        }
    }
    while (Q--) {
        int l, r, k;
        cin >> l >> r >> k;
        vector<int> &arr = lookup[k];
        int i = lower_bound(arr.begin(), arr.end(), l) - arr.begin();
        int j = upper_bound(arr.begin(), arr.end(), r) - arr.begin();
        int ans = j - i;
        cout << ans << endl;
    }
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    solve();
    return 0;
}
```

## D. Anagrams

### Solution 1: sorting, next permutation, dfs, set, backtracking

```cpp
int N, ans;
string S, cur;
set<string> seen;

void dfs(int i) {
    if (i == N) {
        string t = cur;
        sort(t.begin(), t.end());
        if (seen.find(t) != seen.end()) return;
        seen.emplace(t);
        do {
            ans += t.size();
        } while (next_permutation(t.begin(), t.end()));
    }
    for (int j = i; j < N; ++j) {
        cur += S[j];
        dfs(j + 1);
        cur.pop_back();
    }
    dfs(N);
}

void solve() {
    cin >> N >> S;
    ans = 0;
    dfs(0);
    cout << ans << endl;
}

signed main() {
    ios::sync_with_stdio(0);

    cout.tie(0);
```

```cpp
int N, H;
vector<pair<int, int>> monsters;

bool possible(int target) {
    int64 dmg = 0;
```

## F. Chippa Rank

### Solution 1: sorting, offline queries, two pointers

```cpp
struct Motion {
    int d, s;
    Motion() {}
    Motion(int d, int s) : d(d), s(s) {}
    bool operator<(const Motion& other) const {
        return 1LL * d * other.s < 1LL * other.d * s;
    }
};

int N, Q, D;
vector<int> distances, speeds;
vector<pair<int, int>> queries;

void solve() {
    cin >> N >> Q >> D;
    distances.resize(N);
    speeds.resize(N);
    for (int i = 0; i < N; ++i) {
        cin >> distances[i];
        distances[i] += D;
    }
    for (int i = 0; i < N; ++i) {
        cin >> speeds[i];
    }
    vector<Motion> motions(N);
    for (int i = 0; i < N; ++i) {
        motions[i] = Motion(distances[i], speeds[i]);
    };
    sort(motions.begin(), motions.end());
    vector<int> ans(Q);
    for (int i = 0; i < Q; ++i) {
        int x;
        cin >> x;
        queries.emplace_back(x, i);
    }
    sort(queries.rbegin(), queries.rend());
    int i = 0;
    for (const auto &[s, idx] : queries) {
        int d = N + 1 + D;
        while (i < N && 1LL * motions[i].d * s < 1LL * d * motions[i].s) {
            i++;
        }
        ans[idx] = i + 1;
    }
    for (int x : ans) {
        cout << x << endl;
    }
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    solve();
    return 0;
}
```

## G. Tyson's Taunt

### Solution 1: counting inversions, fenwick tree, inequality algebra, rearrange so same terms on same side, forms new prefix sum.

Rearrange average A(r - l + 1) <= S_r - S_(l - 1) <= B(r - l + 1)
F(x) = # {subarrays with avg <= x}
F(b) - F(a), where one is inclusive and other is not. so you calculate F(a) is # {subarrays with avg < A}
S_r - S_(l - 1) <= x(r - l + 1)
s_r - xr <= s_(l - 1) - x(l - 1)
T_i = s_i - x * i
T_r <= T_(l - 1)
count pairs (i, j) 0 <= i < j <= N
T_i >= T_j
For each j, count number of previous i where T_i >= T_j. 
so add into fenwick tree ans query for elements that are greater, and add them in as you move along. 
It's inversion counting cause you say how many elements to the left that are greater than or greater than or equal to current element.

```cpp
int N, A, B;
vector<int> arr;

template <typename T>
struct FenwickTree {
    vector<T> nodes;
    T neutral;

    FenwickTree() : neutral(T(0)) {}

    void init(int n, T neutral_val = T(0)) {
        neutral = neutral_val;
        nodes.assign(n + 1, neutral);
    }

    void update(int idx, T val) {
        while (idx < (int)nodes.size()) {
            nodes[idx] += val;
            idx += (idx & -idx);
        }
    }

    T query(int idx) {
        T result = neutral;
        while (idx > 0) {
            result += nodes[idx];
            idx -= (idx & -idx);
        }
        return result;
    }

    T query(int left, int right) {
        return right >= left ? query(right) - query(left - 1) : T(0);
    }
};

int64 calc(int x, bool inclusive) {
    int64 ans = 0;
    vector<int64> psum(N + 1, 0);
    for (int i = 0; i < N; ++i) {
        psum[i + 1] = psum[i] + arr[i] - x;
    }
    vector<int64> vals = psum;
    sort(vals.begin(), vals.end());
    vals.erase(unique(vals.begin(), vals.end()), vals.end());
    int M = vals.size();
    FenwickTree<int> seg;
    seg.init(M);
    for (int i = 0; i <= N; ++i) {
        int idx = lower_bound(vals.begin(), vals.end(), psum[i]) - vals.begin() + 1;
        ans += seg.query(idx + inclusive, M);
        seg.update(idx, 1);
    }
    return ans;
}

void solve() {
    cin >> N >> A >> B;
    arr.resize(N);
    for (int i = 0; i < N; ++i) {
        cin >> arr[i];
    }

    int64 ans = calc(B, false) - calc(A, true);
    cout << ans << endl;
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    solve();
    return 0;
}
```

## J. Building Bridges

### Solution 1: gcd, union find, max spanning tree

I think this problem is worded wrong, so it should say you need to form a tree, that way you just find max spanning tree with gcd of nodes.

u and v will appear together for every divisor g of w, so you can just loop through the divisors from high to low and merge the components that have vertices divisible by that divisor.

At threshold g, if two DSU components both contain vertices divisible by g, then there exists at least one edge of weight at least g between those components. So merging them now is valid from Kruskal’s perspective.

And if they were already connected earlier, then some combination of edges of weight greater than or equal to g already joined them, so we should skip, again just like Kruskal.

```cpp
const int MAXN = 1e6 + 5;
int N;
vector<int> A;
vector<int> adj[MAXN];

struct UnionFind {
    vector<int> parent, size;
    UnionFind(int n) {
        parent.resize(n);
        iota(parent.begin(),parent.end(),0);
        size.assign(n,1);
    }

    int find(int i) {
        if (i == parent[i]) {
            return i;
        }
        return parent[i] = find(parent[i]);
    }

    void unite(int i, int j) {
        i = find(i), j = find(j);
        if (i != j) {
            if (size[j] > size[i]) {
                swap(i, j);
            }
            size[i] += size[j];
            parent[j] = i;
        }
    }

    bool same(int i, int j) {
        return find(i) == find(j);
    }
};

void solve() {
    cin >> N;
    A.resize(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    for (int i = 0; i < N; ++i) {
        adj[A[i]].emplace_back(i);
    }
    UnionFind dsu(N);
    int64 alpha = 0, beta = -1;
    for (int g = MAXN - 1; g > 0; --g) {
        vector<int> idxs;
        for (int i = g; i < MAXN; i += g) {
            if (!adj[i].empty()) {
                idxs.insert(idxs.end(), adj[i].begin(), adj[i].end());
            }
        }
        for (int i = 1; i < idxs.size(); ++i) {
            if (dsu.same(idxs[0], idxs[i])) continue;
            dsu.unite(idxs[0], idxs[i]);
            alpha += g;
        }
        if (dsu.size[dsu.find(0)] == N) {
            beta = 1LL * g * g;
            break;
        }
    }
    cout << alpha << " " << beta << endl;
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    solve();
    return 0;
}
```
