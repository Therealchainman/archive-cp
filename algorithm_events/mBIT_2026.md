# Montgomery Blair Informatics Tournament 2026

## B. Swapped Coins

### Solution 1: pigeonhole principle, frequency counting, maps

You are given 2N arrays, where each array is the original coin order after exactly one swap. For every coin value x, the algorithm counts how many times x appears at each index.

freq[x][j]++;

The key observation is that the original coin at position i will appear in position i more than N times across the 2N swapped versions. Even though swaps move it sometimes, most swaps do not affect that position.

So for each coin value x, we scan its position-frequency array. If x appears at position i more than N times, then x must be the original coin at index i.

```cpp
int N;
map<int, vector<int>> freq;

void solve() {
    cin >> N;
    for (int i = 0; i < 2 * N; ++i) {
        for (int j = 0; j < N; ++j) {
            int x;
            cin >> x;
            if (freq.find(x) == freq.end()) {
                freq[x] = vector<int>(N, 0);
            }
            freq[x][j]++;
        }
    }
    vector<int> ans(N, 0);
    for (const auto &[x, f] : freq) {
        for (int i = 0; i < N; ++i) {
            if (f[i] > N) {
                ans[i] = x;
                break;
            }
        }
    }
    for (int x : ans) {
        cout << x << " ";
    }
    cout << endl;
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    solve();
    return 0;
}
```

## C. Spiky Sequence

### Solution 1: sweep line piecewise linear, greedy, map counting

```cpp
int N;
vector<int> A;

void solve() {
    cin >> N;
    A.resize(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    int64 cand = 0;
    int dec = 0;
    map<int, int> freq;
    for (int i = 0; i < N; ++i) {
        int cur = min(i + 1, N - i);
        if (A[i] > cur) {
            freq[A[i] - cur]++;
            dec++;
        }
        cand += abs(A[i] - cur);
    }
    int64 ans = cand, base = 0;
    for (auto &[val, count] : freq) {
        int64 delta = val - base;
        cand += delta * (N - dec - dec);
        ans = min(ans, cand);
        dec -= count;
        base = val;
    }
    cout << ans << endl;
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    solve();
    return 0;
}
```

## D. Kevin Skiing

### Solution 1: grid dp, path maximization

```cpp
const int INF = numeric_limits<int>::max();
int R, C;
vector<vector<int>> grid;

void solve() {
    cin >> R >> C;
    grid.resize(R, vector<int>(C));
    for (int r = 0; r < R; r++) {
        for (int c = 0; c < C; c++) {
            cin >> grid[r][c];
        }
    }
    vector<int> dp(C, -INF), ndp(C, -INF);
    if (grid[0][C / 2] != -1) dp[C / 2] = grid[0][C / 2];
    for (int r = 1; r < R; ++r) {
        ndp.assign(C, -INF);
        for (int c = 0; c < C; ++c) {
            if (grid[r][c] == -1) continue;
            for (int dc = -1; dc <= 1; ++dc) {
                int pc = c + dc;
                if (pc < 0 || pc >= C) continue;
                if (dp[pc] == -INF) continue;
                ndp[c] = max(ndp[c], dp[pc] + grid[r][c]);
            }
        }
        swap(dp, ndp);
    }
    int ans = *max_element(dp.begin(), dp.end());
    if (ans == -INF) ans = -1;
    cout << ans << endl;
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    solve();
    return 0;
}
```

## E. Bridge Building

### Solution 1: directed graph, degree counting

You are given two directed graphs with the same number of nodes and edges. The algorithm checks whether every node has the same in-degree and out-degree in both graphs.

For the first graph:

ind1[v]++;
out1[u]++;

For the second graph:

ind2[v]++;
out2[u]++;

Then it compares each node:

if (ind1[i] != ind2[i] || out1[i] != out2[i])

If any node has a different incoming or outgoing count, the graphs cannot satisfy the required condition, so print "no".

Otherwise, print "yes".

The core idea is that the relevant structure of the bridge system is captured by each node’s number of incoming and outgoing bridges, not by the exact matching of edges.

```cpp
int N, M;

void solve() {
    cin >> N >> M;
    vector<int> ind1(N), ind2(N), out1(N), out2(N);
    for (int i = 0; i < M; ++i) {
        int u, v;
        cin >> u >> v;
        --u, --v;
        ind1[v]++;
        out1[u]++;
    }
    for (int i = 0; i < M; ++i) {
        int u, v;
        cin >> u >> v;
        --u, --v;
        ind2[v]++;
        out2[u]++;
    }
    for (int i = 0; i < N; ++i) {
        if (ind1[i] != ind2[i] || out1[i] != out2[i]) {
            cout << "no" << endl;
            return;
        }
    }
    cout << "yes" << endl;
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int T;
    cin >> T;
    while (T--) {
        solve();
    }
    return 0;
}
```

## F. Woololo

### Solution 1: offline queries, sorting, prefix sums, difference array

You have a formula that looks like:

answer(c) = c + sum over adjacent gaps of min(gap², c)

After sorting the values, only the adjacent gaps matter.

For each query c, every gap chooses the cheaper of two costs:

pay gap², if the gap is small
pay c,    if the gap is large

So the problem becomes:

Given many queries c, quickly compute:
Σ min(gap², c)

Instead of doing this from scratch for every query, sort the gaps and sort the queries.

Then as c increases, more gaps become “small enough” that gap² < c. Keep a pointer j for how many gaps are now small, and keep pref, the sum of their gap².

Then:

res = pref + (N - j) * c;

Meaning:

small gaps: pay their actual gap² cost
large gaps + base cost: pay c each

So the algorithm is really:

Sort the gaps, sort the queries, sweep once, and maintain the prefix sum of gaps whose squared cost is cheaper than the query cost.

```cpp
int N, Q;
vector<int> A;
 
void solve() {
    cin >> N >> Q;
    A.assign(N, 0);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    sort(A.begin(), A.end());
    vector<int> diff;
    for (int i = 1; i < N; ++i) {
        diff.emplace_back(A[i] - A[i - 1]);
    }
    sort(diff.begin(), diff.end());
    vector<pair<int, int>> queries;
    for (int i = 0; i < Q; ++i) {
        int c;
        cin >> c;
        queries.emplace_back(c, i);
    }
    vector<int64> ans(Q, 0);
    sort(queries.begin(), queries.end());
    int M = diff.size();
    int64 pref = 0;
    for (int i = 0, j = 0; i < Q; ++i) {
        int c = queries[i].first;
        while (j < M && 1LL * diff[j] * diff[j] < c) {
            pref += 1LL * diff[j] * diff[j];
            ++j;
        }
        int64 res = 1LL * (N - j) * c + pref;
        ans[queries[i].second] = res;
    }
    for (int64 x : ans) {
        cout << x << endl;
    }
}
 
signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int T;
    cin >> T;
    while (T--) {
        solve();
    }
    return 0;
}
```

## G. Swapping Subtrees

### Solution 1: height based heavy decomposition, greedy chain selection

1. Tree decomposition

The tree is decomposed into disjoint downward paths, called chains.

Each node belongs to exactly one chain. A chain starts at some node and repeatedly follows one chosen child until it cannot continue.

In this code, the chosen child is the child with maximum height, meaning the child whose subtree contains the deepest downward path.

if (child_height > best_height) {
    best_height = child_height;
    heavy[u] = v;
}

So for every node u, heavy[u] is the child that lets you continue the longest downward path.

2. Greedy heavy-child choice by height

This is a greedy decomposition.

At each node, the algorithm keeps the tallest child subtree in the same chain. All other child subtrees start new chains.

So the tree is broken into the fewest-looking long vertical pieces by always preserving the longest possible downward continuation.

```cpp
int N;
vector<vector<int>> adj;
vector<int> parent, depth, head, height, chain_len, index_map, heavy;

int heavy_dfs(int u) {
    height[u] = 1;
    heavy[u] = -1;
    int best_height = 0;
    for (int v : adj[u]) {
        if (v == parent[u]) continue;
        parent[v] = u;
        depth[v] = depth[u] + 1;
        int child_height = heavy_dfs(v);
        if (child_height > best_height) {
            best_height = child_height;
            heavy[u] = v;
        }
    }

    height[u] = 1 + best_height;
    return height[u];
}

void decompose(int u, int h) {
    head[u] = h;
    if (u == h) {
        chain_len.emplace_back(0);
    }
    chain_len.back()++;
    if (heavy[u] != -1) {
        decompose(heavy[u], h);
    }
    for (int v : adj[u]) {
        if (v == parent[u]) continue;
        if (v == heavy[u]) continue;
        decompose(v, v);
    }
}

void solve() {
    cin >> N;
    adj.assign(N, vector<int>());
    for (int i = 1; i < N; ++i) {
        int p;
        cin >> p;
        p--;
        adj[p].emplace_back(i);
        adj[i].emplace_back(p);
    }
    parent.assign(N, -1);
    depth.assign(N, 0);
    head.assign(N, 0);
    height.assign(N, 0);
    chain_len.clear();
    index_map.assign(N, 0);
    heavy.assign(N, -1);
    heavy_dfs(0);
    decompose(0, 0);
    int M = chain_len.size();
    sort(chain_len.rbegin(), chain_len.rend());
    vector<int64> pref(M + 1, 0);
    for (int i = 0; i < M; ++i) {
        pref[i + 1] = pref[i] + chain_len[i];
    }
    for (int k = 0; k < N; k++) {
        int chains = k + 1;
        int take = min(chains, M);
        int64 ans = pref[take];
        cout << ans << " ";
    }
    cout << endl;
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int T;
    cin >> T;
    while (T--) {
        solve();
    }
    return 0;
}
```

## H. Watchtower Construction

### Solution 1: greedy, run-length compressed subarray dp, monotonic thresholds, floor-division buckets, block compression

# Key Greedy for One Subarray

Suppose we are fixing one subarray and processing it from right to left.

Let `T` be the maximum allowed size of the next piece to the left.

If we are processing distance:

```cpp
x = d[i]
```

then we need to split `x` so that every piece is at most `T`.

There are two cases.

## Case 1: `x <= T`

No split is needed.

```cpp
pieces = 1
extra_cost = 0
new_threshold = x
```

The whole distance `x` stays as one piece, and the next distance to the left must be at most `x`.

---

## Case 2: `x > T`

We need to split `x`.

The minimum number of pieces needed is:

```cpp
pieces = ceil(x / T)
```

If we split into `pieces` parts, we build:

```cpp
pieces - 1
```

new watchtowers.

So:

```cpp
extra_cost = pieces - 1
```

The new threshold becomes:

```cpp
new_threshold = floor(x / pieces)
```

Why?

Because after splitting `x` into `pieces` nondecreasing positive parts, the best possible leftmost part is at most:

```cpp
floor(x / pieces)
```

That leftmost part is what constrains the next distance to the left.

So the local transition is:

```cpp
pieces = ceil(x / T)
extra_cost = pieces - 1
newT = floor(x / pieces)
```

---

# Why We Process Right to Left

For a subarray to become nondecreasing, each earlier piece must be less than or equal to the next piece on its right.

That means the right side gives a constraint to the left side.

So the natural direction is right to left.

For one subarray:

```cpp
T = d[r]
cost = 0

for i = r - 1 down to l:
    pieces = ceil(d[i] / T)
    cost += pieces - 1
    T = d[i] / pieces
```

This gives the minimum number of new watchtowers for that subarray.

The challenge is that there are `O(n^2)` subarrays, so we cannot run this greedy independently for every subarray.

---

# Main Idea: Process All Subarrays Starting at the Same Left Index Together

Process the distance array from right to left.

At index `i`, we want to represent all subarrays:

```text
d[i ... i]
d[i ... i + 1]
d[i ... i + 2]
...
d[i ... m - 1]
```

where:

```cpp
m = n - 1
```

For every right endpoint `r`, the subarray `d[i ... r]` has:

```text
threshold T[r]
cost f(i, r)
```

The threshold `T[r]` means:

> If we later prepend another distance on the left, this is the maximum allowed piece size that the new distance must fit under.

Instead of storing every endpoint separately, we compress endpoints with the same threshold.

---

# What a Block Means

A block stores:

```cpp
struct Block {
    int threshold;  // shared threshold T
    int cnt;        // number of subarrays/endpoints in this block
    long long cost; // total cost over all subarrays in this block
};
```

A block:

```text
[cnt = 3, threshold = 2, cost = 21]
```

means:

> There are 3 subarrays whose current threshold is `2`, and the total value of `f` over those 3 subarrays is `21`.

It does **not** mean each subarray has cost `21`.

It means the sum of their costs is `21`.

This is enough because the final answer only asks for the total sum.

---

# What Order Are Blocks Stored In?

Blocks are stored in increasing order of right endpoint `r`.

For a fixed left index `i`, the active subarrays are ordered as:

```text
r = i, i + 1, i + 2, ..., m - 1
```

So the blocks compress consecutive ranges of right endpoints.

Example:

```text
r:        4   5   6   7
T[r]:    21   3   3   3
```

This is stored as:

```text
[cnt = 1, threshold = 21]
[cnt = 3, threshold = 3]
```

The first block represents endpoint `r = 4`.

The second block represents endpoints `r = 5, 6, 7`.

---

# Updating Blocks When Adding a New Distance

Suppose we are moving from index `i + 1` to index `i`.

We prepend:

```cpp
x = d[i]
```

For every old block with threshold `T`, all subarrays in that block behave the same way.

Compute:

```cpp
pieces = ceil(x / T)
extra = pieces - 1
newT = x / pieces
```

If the old block was:

```text
[cnt, T, cost]
```

then the updated block becomes:

```text
[cnt, newT, cost + cnt * extra]
```

Why multiply by `cnt`?

Because the block represents `cnt` different subarrays.

If each one needs `extra` new towers after prepending `x`, then the total cost increase for the block is:

```cpp
cnt * extra
```

---

# Adding the New Single-Element Subarray

Every time we process a new index `i`, we also create the subarray:

```text
d[i ... i]
```

This subarray has:

```text
threshold = d[i]
cost = 0
cnt = 1
```

So before the updated old blocks, we add:

```cpp
Block{d[i], 1, 0}
```

This block comes first because it has the smallest right endpoint.

---

# Why Equal Thresholds Are Adjacent

This is one of the most important ideas.

For a fixed left index `i`, as the right endpoint moves farther right, the threshold never increases.

So:

```text
T[i] >= T[i + 1] >= T[i + 2] >= ...
```

Why?

A longer subarray has more constraints from the right side. Adding more distances on the right cannot make the left side easier to fit.

Therefore, thresholds are monotonic in endpoint order.

That means equal thresholds appear in one consecutive run.

You can have:

```text
20 20 10 10 10 3 3 1
```

But you cannot have:

```text
20 10 20 3 10 1
```

because that would violate monotonicity.

This is why merging only adjacent equal blocks is enough.

There is no need to search the whole vector for another block with the same threshold.

---

# Why Blocks Do Not Grow Too Large

This is the subtle part.

At one step, `x = d[i]` is fixed.

Every old threshold `T` is transformed into:

```cpp
pieces = ceil(x / T)
newT = floor(x / pieces)
```

So every new threshold is of the form:

```cpp
floor(x / k)
```

for some integer `k`.

The number of distinct values of:

```cpp
floor(x / k)
```

is only `O(sqrt(x))`.

Example with `x = 25`:

```text
k = 1      -> floor(25 / 1)  = 25
k = 2      -> floor(25 / 2)  = 12
k = 3      -> floor(25 / 3)  = 8
k = 4      -> floor(25 / 4)  = 6
k = 5      -> floor(25 / 5)  = 5
k = 6      -> floor(25 / 6)  = 4
k = 7,8    -> floor(25 / k)  = 3
k = 9..12  -> floor(25 / k)  = 2
k = 13..25 -> floor(25 / k)  = 1
```

Distinct values:

```text
25, 12, 8, 6, 5, 4, 3, 2, 1
```

Only 9 values, not 25.

In general, `floor(x / k)` has at most about `2 * sqrt(x)` distinct values.

Since all updated thresholds must be among those values, and equal thresholds are adjacent, after merging the number of blocks is also `O(sqrt(x))`.

Because the problem has:

```cpp
d[i] <= 100000
```

the number of blocks is bounded by roughly:

```text
2 * sqrt(100000) ≈ 632
```

So iterating over all blocks at every index is fast enough.

---

# Important Clarification: `x` Does Not Need to Be Increasing

The array values `d[i]` can go up and down.

There is no guarantee that `x = d[i]` increases across iterations.

The guarantee is local:

> During one update, `x` is fixed, and the thresholds in the block list are monotonic.

For fixed `x`:

```text
larger T -> smaller pieces count
smaller T -> larger pieces count
```

So if thresholds go down:

```text
T: 20, 4, 2, 1
```

then:

```text
pieces = ceil(x / T)
```

goes up.

For example with `x = 25`:

```text
T = 20 -> pieces = 2
T = 4  -> pieces = 7
T = 2  -> pieces = 13
T = 1  -> pieces = 25
```

Then:

```text
newT = floor(25 / pieces)
```

goes down:

```text
12, 3, 1, 1
```

So the updated thresholds are still monotonic.

That is why the block order stays valid.

---

# Full Algorithm

For each test case:

1. Let `m = n - 1`.
2. Process `d` from right to left.
3. Maintain a vector of compressed blocks.
4. For each `d[i]`:

   * Start a new block for the single subarray `[i, i]`.
   * Update every old block by prepending `d[i]`.
   * Merge adjacent blocks with the same threshold.
   * Add the total block costs to the answer.

# Pattern to Remember

This problem is an example of:

```text
Run-length compressed subarray DP
```

The signs to look for are:

1. The problem asks for a sum over all subarrays or ranges.
2. Each subarray can be represented by a small state.
3. Extending a subarray only depends on that state and the new value.
4. Many states collapse into the same new state.
5. The states are monotonic in endpoint order, so equal states are adjacent.
6. You can store `(state, count, aggregate answer)` instead of storing every subarray.

Similar patterns appear in subarray problems involving:

```text
gcd
bitwise AND
bitwise OR
minimum/maximum
floor-division buckets
monotonic thresholds
```

Here the compressed state is the threshold `T`, and the aggregate answer is the total cost.


```cpp
const int INF = numeric_limits<int>::max(), MOD = 1e9 + 7;

int ceil(int x, int y) {
    return (x + y - 1) / y;
}

int floor(int x, int y) {
    return x / y;
}

struct Block {
    int threshold, cnt;
    int64 cost;
    Block() : threshold(INF), cnt(0), cost(0) {}
    Block(int threshold, int cnt, int64 cost) : threshold(threshold), cnt(cnt), cost(cost) {}
};

int N;
vector<int> D;

void solve() {
    cin >> N;
    int M = N - 1;
    D.assign(M, 0);
    for (int i = 0; i < M; ++i) {
        cin >> D[i];
    }
    int64 ans = 0;
    vector<Block> blocks;
    for (int i = M - 1; i >= 0; --i) {
        vector<Block> newBlocks;
        newBlocks.emplace_back(D[i], 1, 0);
        for (const auto& block : blocks) {
            int newCount = ceil(D[i], block.threshold);
            int add = newCount - 1;
            int newThreshold = floor(D[i], newCount);
            int64 newCost = block.cost + 1LL * add * block.cnt;
            newCost %= MOD;
            if (!newBlocks.empty() && newBlocks.back().threshold == newThreshold) {
                newBlocks.back().cnt += block.cnt;
                newBlocks.back().cost += newCost;
                newBlocks.back().cost %= MOD;
            } else {
                newBlocks.emplace_back(newThreshold, block.cnt, newCost);
            }
        }
        swap(blocks, newBlocks);
        ans += accumulate(blocks.begin(), blocks.end(), 0LL, [](int64 sum, const Block& block) {
            return (sum + block.cost) % MOD;
        });
        ans %= MOD;
    }
    cout << ans << endl;
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int T;
    cin >> T;
    while (T--) {
        solve();
    }
    return 0;
}
```


## I. Collective Castle Coolness

### Solution 1: expected value, uniform random choices, euler phi function, number theory, combinatorics, sum of contributions, 

To start, this problem uses **uniform random choices**, which makes expected value easier to calculate. Since every outcome is equally likely, we can sum all outcomes and divide by the number of outcomes, instead of multiplying each outcome by a different probability.

Another important rule is **linearity of expectation**. Since `S` is the sum of all node coolness values, we can calculate the expected coolness of each node separately and then add them together.

So the reduced problem is: how do we calculate the expected value of one node?

For a node, the only thing that matters is the number of leaves in its subtree. Let that number be `k`. Since the leaf values are assigned by a uniformly random permutation of `1..m`, the values in this node’s subtree form a uniformly random `k`-subset of `1..m`.

So for all subsets of size `k`, we calculate the sum of their gcds, then divide by the total number of `k`-subsets, which is `C(m, k)`.

```text
expected[k] = total_gcd[k] / C(m, k)
```

The hard part is calculating `total_gcd[k]`, the sum of gcds over all `k`-sized subsets of `1..m`.

This is where Euler’s phi function is useful. We use the identity:

```text
n = sum of phi(d) over all d that divide n
```

So every gcd can be expanded into phi terms over its divisors:

```text
gcd(A) = sum of phi(d) over all d that divide gcd(A)
```

Now, `d` divides `gcd(A)` exactly when every number in subset `A` is divisible by `d`.

So for each `d`, we count how many `k`-sized subsets can be formed using only multiples of `d`. There are `floor(m / d)` multiples of `d` from `1..m`, so the number of such subsets is:

```text
C(floor(m / d), k)
```

Each of those subsets contributes a `phi(d)` term to the gcd sum. Therefore:

```text
total_gcd[k] = sum over d from 1 to m of phi(d) * C(floor(m / d), k)
```

Then we divide by `C(m, k)` to get the expected value for any node with `k` leaves, and sum that over all nodes.

Important distinction: this is not counting subsets with gcd exactly `d`. It is counting subsets whose gcd is divisible by `d`, because those are the subsets that include `phi(d)` in their gcd expansion.

1. Rewrite each object’s value as a sum of smaller pieces.
2. Count how many objects each piece contributes to.
3. Sum piece contribution * frequency.

This is very similar in spirit to contribution counting in subarray problems, where instead of iterating over all subarrays, you ask: “How many subarrays does this element contribute to?”

```cpp
const int MAXN = 1e5 + 5, MOD = 998244353;
int phi[MAXN];
int N, M;
vector<vector<int>> adj;
vector<int> leaves;

void precompute_phi(int n) {
    memset(phi, 0, sizeof(phi));
    for (int i = 0; i < n; i++) {
        phi[i] = i;
    }
    for (int i = 2; i < n; i++) {
        if (phi[i] == i) { // i is prime
            for (int j = i; j < n; j += i) {
                phi[j] -= phi[j] / i;
            }
        }
    }
}

int64 inv(int i, int64 m) {
  return i <= 1 ? i : m - (m / i) * inv(m % i, m) % m;
}

vector<int64> fact, inv_fact;

void factorials(int n, int64 m) {
    fact.assign(n + 1, 1);
    inv_fact.assign(n + 1, 0);
    for (int i = 2; i <= n; i++) {
        fact[i] = (fact[i - 1] * i) % m;
    }
    inv_fact.end()[-1] = inv(fact.end()[-1], m);
    for (int i = n - 1; i >= 0; i--) {
        inv_fact[i] = (inv_fact[i + 1] * (i + 1)) % m;
    }
}

int64 choose(int n, int r, int64 m) {
    if (n < r) return 0;
    return (fact[n] * inv_fact[r] % m) * inv_fact[n - r] % m;
}

void dfs(int u, int p = -1) {
    if (adj[u].size() == 1 && p != -1) {
        leaves[u] = 1;
        M++;
    }
    for (int v : adj[u]) {
        if (v == p) continue;
        dfs(v, u);
        leaves[u] += leaves[v];
    }
}

void solve() {
    cin >> N;
    adj.assign(N, vector<int>());
    for (int i = 1; i < N; ++i) {
        int u, v;
        cin >> u >> v;
        u--, v--;
        adj[u].emplace_back(v);
        adj[v].emplace_back(u);
    }
    leaves.assign(N, 0);
    M = 0;
    dfs(0);
    vector<int> total_gcd(M + 1, 0);
    for (int d = 1; d <= M; ++d) {
        int cnt = M / d;
        for (int k = 1; k <= cnt; ++k) {
            total_gcd[k] = (total_gcd[k] + choose(cnt, k, MOD) * phi[d] % MOD) % MOD;
        }
    }
    vector<int> expected(M + 1, 0);
    for (int k = 1; k <= M; ++k) {
        expected[k] = total_gcd[k] * inv(choose(M, k, MOD), MOD) % MOD;
    }
    int64 ans = 0;
    for (int i = 0; i < N; ++i) {
        ans = (ans + expected[leaves[i]]) % MOD;
    }
    cout << ans << endl;
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    precompute_phi(MAXN);
    factorials(MAXN, MOD);
    int T;
    cin >> T;
    while (T--) {
        solve();
    }
    return 0;
}
```
