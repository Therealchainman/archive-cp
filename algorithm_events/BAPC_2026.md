## H. Volanic Islands

### Solution 1: minheap, dijkstra, weighted undirected graph, priority queue

I can solve this problem with dijkstra, minheap algorithm with the lava
And then the same with the sheep king travelling, and checking if there are any deadends, which
is the case if that node became unusable before so if dist[node] < d impossible to continue. 

```cpp
const int64 INF = numeric_limits<int64>::max();
int N, M;
vector<int> A;
vector<vector<pair<int, int>>> adj;

vector<int64> dijkstra() {
    priority_queue<pair<int64, int>, vector<pair<int64, int>>, greater<pair<int64, int>>> minheap;
    vector<int64> ans(N, INF);
    for (int i = 0; i < N; ++i) {
        minheap.emplace(A[i], i);
        ans[i] = A[i];
    }
    while (!minheap.empty()) {
        auto [d, u] = minheap.top();
        minheap.pop();
        if (d > ans[u]) continue;
        for (auto [v, w] : adj[u]) {
            if (d + w < ans[v]) {
                ans[v] = d + w;
                minheap.emplace(ans[v], v);
            }
        }
    }
    return ans;
}

void solve() {
    cin >> N >> M;
    A.assign(N, 0);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    adj.assign(N, vector<pair<int, int>>());
    for (int i = 0; i < M; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        --u, --v;
        adj[u].emplace_back(v, w);
        adj[v].emplace_back(u, w);
    }
    vector<int64> dist = dijkstra();
    priority_queue<pair<int64, int>, vector<pair<int64, int>>, greater<pair<int64, int>>> minheap;
    vector<int64> dp(N, INF);
    minheap.emplace(0, 0);
    while (!minheap.empty()) {
        auto [d, u] = minheap.top();
        minheap.pop();
        if (d > dist[u] || d > dp[u]) continue;
        if (u == N - 1) {
            cout << "YES" << endl;
            return;
        }
        for (auto [v, w] : adj[u]) {
            if (d + w < dp[v]) {
                dp[v] = d + w;
                minheap.emplace(dp[v], v);
            }
        }
    }
    cout << "NO" << endl;
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int T;
    cin >> T;
    while (T--) {
        solve();
    }
    return 0;
}
```
