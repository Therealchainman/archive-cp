### Solution 1: greedy, binary search

```cpp
int N, K, X;
vector<int> A;

bool possible(int target) {
    int cnt = 0;
    for (int i = 0, p = 0; cnt < K && p <= X;) {
        if (i == N || A[i] - p >= target) {
            cnt++;
            p++;
        } else {
            p = max(p, A[i++] + target);
        }
    }
    return cnt >= K;
}

void solve() {
    cin >> N >> K >> X;
    A.assign(N, 0);
    for (int i = 0; i < N; i++) {
        cin >> A[i];
    }
    sort(A.begin(), A.end());
    int lo = 0, hi = 1e9;
    while (lo < hi) {
        int mid = lo + (hi - lo + 1) / 2;
        if (possible(mid)) {
            lo = mid;
        } else {
            hi = mid - 1;
        }
    }
    // generate output
    vector<int> ans;
    for (int i = 0, p = 0, cnt = 0; cnt < K && p <= X;) {
        if (i == N || A[i] - p >= lo) {
            cnt++;
            ans.emplace_back(p++);
        } else {
            p = max(p, A[i++] + lo);
        }
    }
    for (int x : ans) {
        cout << x << " ";
    }
    cout << endl;
}
```

### Solution 1: dynamic programming, max cost weakly increasing subsequence, LIS related

```cpp
int N;
vector<int> A, C;

void solve() {
    cin >> N;
    A.assign(N, 0);
    C.assign(N, 0);
    for (int i = 0; i < N; i++) {
        cin >> A[i];
    }
    vector<int64> dp(N, 0);
    int64 total = 0;
    for (int i = 0; i < N; i++) {
        cin >> C[i];
        dp[i] = C[i];
        total += C[i];
    }
    for (int i = 1; i < N; ++i) {
        for (int j = 0; j < i; ++j) {
            if (A[i] < A[j]) continue;
            dp[i] = max(dp[i], dp[j] + C[i]);
        }
    }
    int64 ans = total - *max_element(dp.begin(), dp.end());
    cout << ans << endl;
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int T;
    cin >> T;
    while (T--) {
        solve();
    }
    return 0;
}
```
